## 使用範例

### 取得指標

```rust
use std::sync::atomic::Ordering;
use crate::msgq::AsyncQueue;

let queue = AsyncQueue::new("orders");
let metrics = queue.metrics();

// 放入一些訊息
queue.enqueue(b"order-1".to_vec(), 30).await?;
queue.enqueue(b"order-2".to_vec(), 30).await?;

// 讀取指標
println!("入隊總數: {}", metrics.enqueued_total.load(Ordering::Relaxed));
println!("佇列深度: {}", metrics.queue_depth.load(Ordering::Relaxed));

// 取出訊息
let msg = queue.dequeue(0).await?.unwrap();
println!("出隊總數: {}", metrics.dequeued_total.load(Ordering::Relaxed));
println!("處理中: {}", metrics.in_flight.load(Ordering::Relaxed));

// 確認處理
queue.ack(&msg.id).await?;
println!("確認總數: {}", metrics.acked_total.load(Ordering::Relaxed));
```

### 健康檢查

```rust
let health = queue.health().await;

match &health.status {
    HealthStatus::Healthy => {
        println!("狀態: 正常");
        if !health.details.is_empty() {
            println!("詳情: {:?}", health.details);
        }
    }
    HealthStatus::Degraded(reason) => {
        println!("狀態: 降級 ({})", reason);
    }
    HealthStatus::Unhealthy(reason) => {
        println!("狀態: 異常 ({})", reason);
    }
}
```

### 健康閾值

```rust
impl AsyncQueue {
    pub async fn health(&self) -> QueueHealth {
        let depth = self.length().await.unwrap_or(0);
        let inflight = {
            let inflight = self.inflight.read().await;
            inflight.len() as u64
        };

        if depth > 10000 {
            QueueHealth::unhealthy("queue depth exceeds 10000")
        } else if inflight > 5000 {
            QueueHealth::degraded("high in-flight message count")
        } else {
            let mut details = HashMap::new();
            details.insert("depth".to_string(), depth.to_string());
            details.insert("inflight".to_string(), inflight.to_string());
            QueueHealth {
                status: HealthStatus::Healthy,
                details,
            }
        }
    }
}
```

閾值說明：
- `depth > 10000` → Unhealthy
- `inflight > 5000` → Degraded
- 其他 → Healthy
