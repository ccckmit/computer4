## API 設計

```rust
use futures::stream::Stream;

pub struct AsyncQueueStream {
    queue: AsyncQueue,
    buffer: VecDeque<AsyncQueueMessage>,
}

impl Stream for AsyncQueueStream {
    type Item = Result<AsyncQueueMessage, String>;
}
```

## 使用範例

```rust
// 建立佇列並加入訊息
let mut queue = AsyncQueue::new("test");
queue.enqueue(b"msg1".to_vec(), 30).await?;
queue.enqueue(b"msg2".to_vec(), 30).await?;
queue.enqueue(b"msg3".to_vec(), 30).await?;

// 使用 for await 迭代
let mut stream = queue.stream();
while let Some(result) = stream.next().await {
    match result {
        Ok(msg) => {
            println!("收到訊息: {:?}", msg.payload_str());
            // 處理後確認
            queue.ack(&msg.id).await?;
        }
        Err(e) => {
            eprintln!("錯誤: {}", e);
        }
    }
}

// 或使用 while let
for await in queue.stream() {
    let msg = in?;
    println!("{:?}", msg);
}
```
