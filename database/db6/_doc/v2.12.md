## API 使用

```rust
use db6::kv::{KvStore, KvEngine};

// 記憶體模式
let mut kv = KvEngine::new("memory")?;  // HashMemoryEngine
let mut kv = KvEngine::new("btree")?;  // BTreeMemoryEngine
let mut kv = KvEngine::new("lsm")?;     // LsmEngine

// 持久化模式
let mut kv = KvEngine::open("btree", "/path/to/db")?;
let mut kv = KvEngine::open("lsm", "/path/to/db")?;

// 操作
kv.put(1, b"key", b"value")?;
let val = kv.get(1, b"key")?;
kv.flush()?;
```
