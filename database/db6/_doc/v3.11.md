# v3.11 - JOIN 支援

## 日期：2026-05-17

## 目標

在 SQL Executor 中支援 JOIN 操作，讓 SQL 語法能夠關聯多個表格的資料。

## 背景

### 現有架構

```rust
// query/ API 已有查詢功能
db.select("*")
    .from("users")
    .join("orders", "users.id", "orders.user_id")
    .execute();

// SQL API 原本不支援 JOIN
SELECT * FROM users u JOIN orders o ON u.id = o.user_id;  // 不支援
```

### v3.11 目標：SQL 支援 JOIN

```sql
-- INNER JOIN：只返回匹配的列
SELECT * FROM users u JOIN orders o ON u.id = o.user_id;

-- LEFT JOIN：返回左表所有列，，右表沒有匹配則為 NULL
SELECT * FROM users u LEFT JOIN orders o ON u.id = o.user_id;
```

## 語法設計

### INNER JOIN

```sql
-- 基本語法
SELECT * FROM users u JOIN orders o ON u.id = o.user_id;

-- 多表 JOIN
SELECT * FROM users u
    JOIN orders o ON u.id = o.user_id
    JOIN products p ON o.product_id = p.id;

-- USING 語法
SELECT * FROM users u JOIN orders o USING (user_id);

-- 帶 WHERE 條件
SELECT * FROM users u JOIN orders o ON u.id = o.user_id WHERE u.age > 25;
```

### LEFT JOIN

```sql
-- 基本語法
SELECT * FROM users u LEFT JOIN orders o ON u.id = o.user_id;

-- 左表保留所有列，右表沒有匹配為空
SELECT u.name, o.order_id FROM users u LEFT JOIN orders o ON u.id = o.user_id;
```

### CROSS JOIN

```sql
-- 笛卡爾積（暫不支援）
SELECT * FROM users CROSS JOIN orders;
```

## API 設計

### SQL 語法

```sql
-- 簡單 JOIN
SELECT u.name, o.amount
FROM users u
JOIN orders o ON u.id = o.user_id;

-- 多條件 JOIN
SELECT *
FROM users u
JOIN orders o ON u.id = o.user_id AND o.amount > 100;

-- 聚合與 JOIN
SELECT u.city, COUNT(*) as cnt
FROM users u
LEFT JOIN orders o ON u.id = o.user_id
GROUP BY u.city;
```

### Rust API 對照

| query/ API | SQL API |
|------------|---------|
| `.join("orders", "user_id", "user_id")` | `JOIN orders o ON u.user_id = o.user_id` |
| `.join_with_condition(...)` | `JOIN orders o ON condition` |
| `.left_join(...)` | `LEFT JOIN orders o ON ...` |

```rust
// src/sql/parser/ast.rs

pub struct Join {
    pub kind: JoinKind,
    pub table: TableRef,
    pub condition: JoinCondition,
}

pub enum JoinKind {
    Inner, Left, Right, Full, Cross, Natural,
}

pub enum JoinCondition {
    On(Expr),           // ON expr
    Using(Vec<String>), // USING (col1, col2)
    None,               // CROSS JOIN
}
```

### 3. Executor 實作

```rust
fn execute_select(&mut self, select: &SelectStmt) -> Result<ResultSet> {
    // 1. 從 FROM 取得主表
    let from_item = select.from.as_ref().ok_or(...)?;
    let main_table = get_table_name(from_item);

    // 2. 掃描主表
    let main_prefix = format!("{}:", main_table);
    let mut rows = self.engine.scan(1, main_prefix.as_bytes(), ...)?;

    // 3. 處理每個 JOIN
    for join in &select.joins {
        let right_table = &join.table.name;
        let right_rows = self.engine.scan(1, ...)?;

        match &join.condition {
            JoinCondition::On(expr) => {
                match join.kind {
                    JoinKind::Inner => {
                        rows = inner_join(rows, right_rows, expr);
                    }
                    JoinKind::Left => {
                        rows = left_join(rows, right_rows, expr);
                    }
                    _ => return Err("Unsupported JOIN type"),
                }
            }
            JoinCondition::Using(cols) => {
                // 轉換為 ON 條件
            }
            JoinCondition::None => {
                // CROSS JOIN - 笛卡爾積
            }
        }
    }

    // 4. 後續處理（WHERE, GROUP BY, ORDER BY, LIMIT）
    ...
}

fn inner_join(
    left_rows: Vec<(Vec<u8>, Vec<u8>)>,
    right_rows: Vec<(Vec<u8>, Vec<u8>)>,
    condition: &Expr,
) -> Vec<(Vec<u8>, Vec<u8>)> {
    let mut result = Vec::new();

    for (lk, lv) in &left_rows {
        for (rk, rv) in &right_rows {
            if eval_join_condition(condition, lk, lv, rk, rv) {
                // 合併列 - 通常是 lv 或自訂格式
                result.push((lk.clone(), lv.clone()));
            }
        }
    }

    result
}

fn left_join(
    left_rows: Vec<(Vec<u8>, Vec<u8>)>,
    right_rows: Vec<(Vec<u8>, Vec<u8>)>,
    condition: &Expr,
) -> Vec<(Vec<u8>, Vec<u8>)> {
    let mut result = Vec::new();

    for (lk, lv) in &left_rows {
        let mut matched = false;
        for (rk, rv) in &right_rows {
            if eval_join_condition(condition, lk, lv, rk, rv) {
                result.push((lk.clone(), lv.clone()));
                matched = true;
            }
        }
        if !matched {
            // 左表保留，右表為 NULL
            result.push((lk.clone(), b"null".to_vec()));
        }
    }

    result
}
```

### 4. Join 條件評估

```rust
fn eval_join_condition(
    expr: &Expr,
    left_key: &[u8],
    left_val: &[u8],
    right_key: &[u8],
    right_val: &[u8],
) -> bool {
    match expr {
        Expr::BinOp { left, op: BinOp::Eq, right } => {
            let left_val = extract_column_value(left, left_key, left_val);
            let right_val = extract_column_value(right, right_key, right_val);
            left_val == right_val
        }
        Expr::BinOp { left, op: BinOp::And, right } => {
            eval_join_condition(left, left_key, left_val, right_key, right_val)
                && eval_join_condition(right, left_key, left_val, right_key, right_val)
        }
        // ...
    }
}

fn extract_column_value(expr: &Expr, key: &[u8], val: &[u8]) -> String {
    match expr {
        Expr::Column { name: "key", .. } => String::from_utf8_lossy(key).to_string(),
        Expr::Column { name, .. } => json_path_get(&String::from_utf8_lossy(val), name),
        Expr::JsonPath { path, .. } => json_path_get(&String::from_utf8_lossy(val), path),
        // ...
    }
}
```

## 引擎限制

### 可用引擎

| 引擎 | JOIN 支援 |
|------|----------|
| BTreeEngine | ✅ 支援 |
| BTreeMemoryEngine | ✅ 支援 |
| HashMemoryEngine | ❌ 不支援 |
| LsmEngine | ❌ 不支援（高 I/O 成本） |

### Capability 檢查

```rust
pub trait CanJoin: StorageEngine {}

// 只有 BTree 系列引擎實作 CanJoin
impl CanJoin for BTreeEngine {}
impl CanJoin for BTreeMemoryEngine {}
```

## 測試案例

### 基本 INNER JOIN

```rust
#[test]
fn test_inner_join() {
    let mut exec = Executor::new(Box::new(BTreeMemoryEngine::new()));

    // 插入 users
    exec.execute("INSERT INTO users VALUES ('1', '{\"name\":\"Alice\"}')").unwrap();
    exec.execute("INSERT INTO users VALUES ('2', '{\"name\":\"Bob\"}')").unwrap();

    // 插入 orders
    exec.execute("INSERT INTO orders VALUES ('1', '{\"user_id\":\"1\",\"amount\":100}')").unwrap();
    exec.execute("INSERT INTO orders VALUES ('2', '{\"user_id\":\"2\",\"amount\":200}')").unwrap();
    exec.execute("INSERT INTO orders VALUES ('3', '{\"user_id\":\"1\",\"amount\":150}')").unwrap();

    let result = exec.execute(
        "SELECT * FROM users u JOIN orders o ON u.key = o.user_id"
    ).unwrap();

    // 應該有 3 行（Alice 有 2 筆訂單，Bob 有 1 筆）
    assert_eq!(result.rows.len(), 3);
}
```

### 多表 JOIN

```rust
#[test]
fn test_multiple_joins() {
    let mut exec = Executor::new(Box::new(BTreeMemoryEngine::new()));

    exec.execute("INSERT INTO users VALUES ('1', '{\"name\":\"Alice\"}')").unwrap();
    exec.execute("INSERT INTO orders VALUES ('1', '{\"user_id\":\"1\",\"product_id\":\"A\"}')").unwrap();
    exec.execute("INSERT INTO products VALUES ('A', '{\"name\":\"Widget\"}')").unwrap();

    let result = exec.execute(
        "SELECT u.name, p.name FROM users u
         JOIN orders o ON u.key = o.user_id
         JOIN products p ON o.product_id = p.key"
    ).unwrap();

    assert_eq!(result.rows.len(), 1);
}
```

## 不做的事項

- ❌ CROSS JOIN（笛卡爾積）- 效能風險太高
- ❌ FULL OUTER JOIN - 實作複雜度
- ❌ NATURAL JOIN - 自動欄位匹配有歧義
- ❌ JOIN 最佳化（巢狀迴圈 vs Hash Join vs Merge Join）
- ❌ JOIN 與索引配合

## 附錄：完整範例

```sql
-- 建立表格
CREATE TABLE users;
CREATE TABLE orders;
CREATE TABLE products;

-- 插入資料
INSERT INTO users VALUES ('1', '{"name":"Alice","city":"Taipei"}');
INSERT INTO users VALUES ('2', '{"name":"Bob","city":"Kaohsiung"}');
INSERT INTO users VALUES ('3', '{"name":"Charlie","city":"Taipei"}');

INSERT INTO orders VALUES ('101', '{"user_id":"1","product_id":"A","amount":100}');
INSERT INTO orders VALUES ('102', '{"user_id":"1","product_id":"B","amount":200}');
INSERT INTO orders VALUES ('103', '{"user_id":"2","product_id":"A","amount":150}');

INSERT INTO products VALUES ('A', '{"name":"Widget","price":50}');
INSERT INTO products VALUES ('B', '{"name":"Gadget","price":100}');

-- INNER JOIN：只返回有訂單的使用者
SELECT u.name, o.amount
FROM users u
JOIN orders o ON u.key = o.user_id;

-- LEFT JOIN：保留所有使用者
SELECT u.name, o.amount
FROM users u
LEFT JOIN orders o ON u.key = o.user_id;

-- 多表 JOIN
SELECT u.name, p.name as product, o.amount
FROM users u
JOIN orders o ON u.key = o.user_id
JOIN products p ON o.product_id = p.key;

-- 帶聚合的 JOIN
SELECT u.city, COUNT(*) as order_count, SUM(o.amount) as total
FROM users u
LEFT JOIN orders o ON u.key = o.user_id
GROUP BY u.city;
```
