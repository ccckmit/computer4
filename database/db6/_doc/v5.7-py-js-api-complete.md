## Target API

### Python (db6py)

```python
from db6py import Client

client = Client("http://localhost:50052")

# KV operations
client.put(1, "key", "value")
value, found = client.get(1, "key")
client.delete(1, "key")
client.batch_put(1, [("k1", "v1"), ("k2", "v2")])
pairs = client.scan(1, "k", "k~")
client.range_delete(1, "k1", "k2")
stats = client.stats()

# SQL
result = client.execute_sql("SELECT * FROM users WHERE age > 18")
# result = {columns: [...], rows: [[...], ...], affected: 0}

# FTS
client.fts_insert(1, "Hello world")
client.fts_insert(2, "資料庫系統")
doc_ids = client.fts_search("world")
# doc_ids = [1]
results = client.fts_search_bm25("database")
# results = [{doc_id: 1, score: 2.5}, ...]

# Queue
client.queue_create("myqueue")
client.queue_enqueue("myqueue", "hello", timeout=30)
msg = client.queue_dequeue("myqueue", wait_timeout=5)
# msg = {msg_id: "...", payload: "...", delivery_count: 1}
client.queue_ack("myqueue", msg["msg_id"])
stats = client.queue_stats("myqueue")
queues = client.queue_list()

# PubSub
client.pubsub_publish("channel1", "hello")
```

### Node.js (db6nodejs)

```javascript
import { Client } from 'db6nodejs';

const client = new Client('http://localhost:50052');

// KV operations
await client.put(1, 'key', 'value');
const [value, found] = await client.get(1, 'key');
await client.delete(1, 'key');
await client.batchPut(1, [['k1', 'v1'], ['k2', 'v2']]);
const pairs = await client.scan(1, 'k', 'k~');
await client.rangeDelete(1, 'k1', 'k2');
const stats = await client.stats();

// SQL
const result = await client.executeSql('SELECT * FROM users WHERE age > 18');
// result = {columns: [...], rows: [[...], ...], affected: 0}

// FTS
await client.ftsInsert(1, 'Hello world');
await client.ftsInsert(2, '資料庫系統');
const docIds = await client.ftsSearch('world');
// docIds = [1]
const results = await client.ftsSearchBm25('database');
// results = [{doc_id: 1, score: 2.5}, ...]

// Queue
await client.queueCreate('myqueue');
await client.queueEnqueue('myqueue', 'hello', 30);
const msg = await client.queueDequeue('myqueue', 5);
// msg = {msg_id: '...', payload: '...', delivery_count: 1}
await client.queueAck('myqueue', msg.msg_id);
const stats = await client.queueStats('myqueue');
const queues = await client.queueList();

// PubSub
await client.pubsubPublish('channel1', 'hello');
```
