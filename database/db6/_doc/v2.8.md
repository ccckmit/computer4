## API 使用

```rust
// BTree 持久化
let mut engine = BTreeEngine::open(Path::new("/path/to/db"))?;
engine.put(1, b"key", b"value")?;
engine.flush()?;

// LSM 持久化
let mut engine = LsmEngine::open(Path::new("/path/to/db"))?;
engine.put(1, b"key", b"value")?;
engine.flush()?;  // 寫入 SSTable

// LSM WAL Recovery (自動)
let engine = LsmEngine::open(Path::new("/path/to/db"))?;
// 透過交易寫入的資料會自動從 WAL 恢復
```
